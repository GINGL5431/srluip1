                mobileMenu.classList.remove('opacity-0', '-translate-y-full', 'pointer-events-none');
                mobileMenu.classList.add('opacity-100', 'translate-y-0', 'pointer-events-auto');
                menuBtn.innerHTML = '<i class="fa fa-times"></i>';
            } else {
                mobileMenu.classList.add('opacity-0', '-translate-y-full', 'pointer-events-none');
                mobileMenu.classList.remove('opacity-100', 'translate-y-0', 'pointer-events-auto');
                menuBtn.innerHTML = '<i class="fa fa-bars"></i>';
            }
        });

                e.preventDefault();

                    menuBtn.innerHTML = '<i class="fa fa-bars"></i>';

        const groupModal = document.getElementById('groupModal');
        const closeModal = document.getElementById('closeModal');
        const playButtons = [
            document.getElementById('playBtn'),
            document.getElementById('mobilePlayBtn'),
            document.getElementById('heroPlayBtn'),
            document.getElementById('ctaPlayBtn')
        ];
        
        function openModal() {
            groupModal.classList.remove('opacity-0', 'pointer-events-none');
            groupModal.classList.add('opacity-100', 'pointer-events-auto');
            groupModal.querySelector('div').classList.remove('scale-95');
            groupModal.querySelector('div').classList.add('scale-100');
            document.body.style.overflow = 'hidden';
        }
        
        function closeModalFunc() {
            groupModal.classList.add('opacity-0', 'pointer-events-none');
            groupModal.classList.remove('opacity-100', 'pointer-events-auto');
            groupModal.querySelector('div').classList.add('scale-95');
            groupModal.querySelector('div').classList.remove('scale-100');
            document.body.style.overflow = '';
        }
        
        playButtons.forEach(btn => {
            if (btn) {
                btn.addEventListener('click', openModal);
            }
        });
        
        closeModal.addEventListener('click', closeModalFunc);
        
        // 点击模态框外部关闭
        groupModal.addEventListener('click', (e) => {
            if (e.target === groupModal) {
                closeModalFunc();
            }
        });
        
        // 按ESC键关闭
        document.addEventListener('keydown', (e) => {
            if (e.key === 'Escape' && !groupModal.classList.contains('opacity-0')) {
                closeModalFunc();
            }
        });
    </script>
</body>
</html>
